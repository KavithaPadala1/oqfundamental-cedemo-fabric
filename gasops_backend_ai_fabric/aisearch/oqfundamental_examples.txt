-- Comparison of tasks for 16inch Electrofusion vs Excavation and Backfill with ticks
WITH TasksCTE AS (
    SELECT DISTINCT
        DescriptionName,
        UPPER(LTRIM(RTRIM(value))) AS Task
    FROM vm_cedemo_oqrequirements
    CROSS APPLY STRING_SPLIT(TaskCodesAndDescriptions, ';')
    WHERE DescriptionName IN ('16inch Electrofusion', 'Excavation and Backfill')
      AND LTRIM(RTRIM(value)) <> ''
),
TaskPresence AS (
    SELECT 
        Task,
        MAX(CASE WHEN DescriptionName = '16inch Electrofusion' THEN 1 ELSE 0 END) AS Electrofusion,
        MAX(CASE WHEN DescriptionName = 'Excavation and Backfill' THEN 1 ELSE 0 END) AS Excavation
    FROM TasksCTE
    GROUP BY Task
)
-- Tick table
SELECT 
    Task,
    CASE WHEN Electrofusion = 1 THEN '✓' ELSE '✗' END AS [In 16inch Electrofusion],
    CASE WHEN Excavation = 1 THEN '✓' ELSE '✗' END AS [In Excavation and Backfill]
FROM TaskPresence
ORDER BY Task;
-- Count summary of tasks for 16inch Electrofusion vs Excavation and Backfill
WITH TasksCTE AS (
    SELECT DISTINCT
        DescriptionName,
        UPPER(LTRIM(RTRIM(value))) AS Task
    FROM vm_cedemo_oqrequirements
    CROSS APPLY STRING_SPLIT(TaskCodesAndDescriptions, ';')
    WHERE DescriptionName IN ('16inch Electrofusion', 'Excavation and Backfill')
      AND LTRIM(RTRIM(value)) <> ''
),
TaskPresence AS (
    SELECT 
        Task,
        MAX(CASE WHEN DescriptionName = '16inch Electrofusion' THEN 1 ELSE 0 END) AS Electrofusion,
        MAX(CASE WHEN DescriptionName = 'Excavation and Backfill' THEN 1 ELSE 0 END) AS Excavation
    FROM TasksCTE
    GROUP BY Task
)
SELECT
    SUM(Electrofusion) AS TotalTasks_16inchElectrofusion,
    SUM(Excavation) AS TotalTasks_ExcavationAndBackfill,

    SUM(CASE WHEN Electrofusion = 0 AND Excavation = 1 THEN 1 ELSE 0 END) AS TasksOnlyInExcavationAndBackfill
FROM TaskPresence;

User : show me employee operator qualifications 
User: show me employee OQ status
SQL:
WITH EmployeeRoles AS (
    SELECT
        ce.EmployeeMasterID,
        ce.EmployeeName,
        ce.EmployeeNumber,
        TRIM(value) AS RoleName
    FROM vm_cedemo_companyemployees_active ce
    CROSS APPLY STRING_SPLIT(ce.ITSRoleNames, ';')
    WHERE ce.IsActive = 1
),
RequiredWAF AS (
    SELECT
        er.EmployeeMasterID,
        er.EmployeeName,
        er.EmployeeNumber,
        er.RoleName,
        TRY_CAST(TRIM(waf.value) AS BIGINT) AS WorkActivityFunctionID
    FROM EmployeeRoles er
    LEFT JOIN vm_cedemo_oqrequirements oq
        ON oq.TypeName = 'Company ITS Role'
        AND oq.DescriptionName = er.RoleName
    CROSS APPLY STRING_SPLIT(ISNULL(oq.WorkActivityFunctionIDs, ''), ';') waf
),
RequiredCount AS (
    SELECT
        EmployeeMasterID,
        EmployeeNumber,
        EmployeeName,
        RoleName,
        COUNT(DISTINCT WorkActivityFunctionID) AS TotalRequiredWAF
    FROM RequiredWAF
    WHERE WorkActivityFunctionID IS NOT NULL
    GROUP BY EmployeeMasterID, EmployeeNumber, EmployeeName, RoleName
),
QualifiedWAF AS (
    SELECT
        rw.EmployeeMasterID,
        rw.EmployeeNumber,
        rw.RoleName,
        COUNT(DISTINCT q.WorkActivityFunctionID) AS QualifiedCount
    FROM RequiredWAF rw
    INNER JOIN vm_cedemo_CompanyEmployeesWAFQualifications q
        ON rw.WorkActivityFunctionID = q.WorkActivityFunctionID
        AND rw.EmployeeMasterID = q.EmployeeMasterID
    WHERE UPPER(TRIM(q.Qualified)) = 'YES'
    GROUP BY rw.EmployeeMasterID, rw.EmployeeNumber, rw.RoleName
),
RoleQualification AS (
    SELECT
        rc.EmployeeMasterID,
        rc.EmployeeNumber,
        rc.EmployeeName,
        rc.RoleName,
        CASE 
            WHEN rc.TotalRequiredWAF IS NULL OR rc.TotalRequiredWAF = 0 THEN 'YES'
            WHEN qc.QualifiedCount = rc.TotalRequiredWAF THEN 'YES'
            ELSE 'NO'
        END AS Qualified
    FROM RequiredCount rc
    LEFT JOIN QualifiedWAF qc
        ON rc.EmployeeMasterID = qc.EmployeeMasterID
        AND rc.RoleName = qc.RoleName
),
EmployeeSummary AS (
    SELECT
        ce.EmployeeMasterID,
        ce.EmployeeNumber,
        ce.EmployeeName,
        -- Total roles assigned
        COUNT(DISTINCT rq.RoleName) AS TotalRoles,
        SUM(CASE WHEN rq.Qualified = 'YES' THEN 1 ELSE 0 END) AS QualifiedRoles,
        SUM(CASE WHEN rq.Qualified = 'NO' THEN 1 ELSE 0 END) AS NotQualifiedRoles,
        -- Direct check if employee has ITSRoleNames
        CASE 
            WHEN ce.ITSRoleNames IS NULL OR LTRIM(RTRIM(ce.ITSRoleNames)) = '' THEN 'NO'
            ELSE 'YES'
        END AS HasRoles
    FROM vm_cedemo_companyemployees_active ce
    LEFT JOIN RoleQualification rq
        ON ce.EmployeeMasterID = rq.EmployeeMasterID
    WHERE ce.IsActive = 1
    GROUP BY ce.EmployeeMasterID, ce.EmployeeNumber, ce.EmployeeName, ce.ITSRoleNames
)
SELECT
    COUNT(*) AS TotalEmployees,
    SUM(CASE WHEN HasRoles = 'NO' THEN 1 ELSE 0 END) AS EmployeesWithoutRoles,
    SUM(CASE WHEN HasRoles = 'YES' THEN 1 ELSE 0 END) AS EmployeesWithRoles,
    SUM(CASE WHEN NotQualifiedRoles = 0 AND HasRoles = 'YES' THEN 1 ELSE 0 END) AS FullyQualifiedEmployees,
    SUM(CASE WHEN NotQualifiedRoles > 0 AND HasRoles = 'YES' THEN 1 ELSE 0 END) AS EmployeesWithMissingQualifications
FROM EmployeeSummary;

User : show me qualified contractor employees for contractor CAC
SQL:
-- Step 1: Get contractor employees with roles
WITH ContractorEmployees AS (
    SELECT DISTINCT
        EmployeeMasterID,
        EmployeeName,
        TRIM(value) AS RoleName
    FROM vm_cedemo_contractoremployees_active
    CROSS APPLY STRING_SPLIT(ITSRoleNames, ';')
    WHERE IsActive = 1
      AND (ContractorName LIKE '%CAC%' OR ContractorDisplayName LIKE '%CAC%')
      AND TRIM(ISNULL(ITSRoleNames,'')) <> ''
      AND TRIM(value) <> ''
),
-- Step 2: Required WAFs for roles
RoleRequiredWAF AS (
    SELECT
        DescriptionName AS RoleName,
        TRY_CAST(TRIM(value) AS BIGINT) AS WorkActivityFunctionID
    FROM vm_cedemo_oqrequirements
    CROSS APPLY STRING_SPLIT(WorkActivityFunctionIDs, ';')
    WHERE TRIM(value) <> ''
),
-- Step 3: Employee + required WAF mapping
EmployeeRoleRequirements AS (
    SELECT
        ce.EmployeeMasterID,
        ce.EmployeeName,
        ce.RoleName,
        rw.WorkActivityFunctionID
    FROM ContractorEmployees ce
    LEFT JOIN RoleRequiredWAF rw
        ON ce.RoleName = rw.RoleName
),
-- Step 4: Qualification check
QualificationCheck AS (
    SELECT
        err.EmployeeMasterID,
        err.EmployeeName,
        err.RoleName,
        err.WorkActivityFunctionID,
        CASE
            WHEN q.WorkActivityFunctionID IS NOT NULL
                 AND q.Qualified = 'Yes'
            THEN 1
            ELSE 0
        END AS QualifiedFlag
    FROM EmployeeRoleRequirements err
    LEFT JOIN vm_cedemo_ContractorEmployeesWAFQualifications q
        ON err.EmployeeMasterID = q.EmployeeMasterID
        AND err.WorkActivityFunctionID = q.WorkActivityFunctionID
),
-- Step 5: Determine role qualification status
RoleStatus AS (
    SELECT
        EmployeeMasterID,
        EmployeeName,
        RoleName,
        COUNT(*) AS RequiredWAFCount,
        SUM(QualifiedFlag) AS QualifiedWAFCount,
        CASE
            WHEN COUNT(*) = SUM(QualifiedFlag)
            THEN 'Qualified'
            ELSE 'Not Qualified'
        END AS RoleQualificationStatus
    FROM QualificationCheck
    GROUP BY EmployeeMasterID, EmployeeName, RoleName
)
-- Step 6: Employee summary
SELECT
    EmployeeMasterID,
    EmployeeName,
    COUNT(RoleName) AS TotalRoles,
    SUM(CASE WHEN RoleQualificationStatus = 'Qualified' THEN 1 ELSE 0 END)
        AS QualifiedRoleCount,
    SUM(CASE WHEN RoleQualificationStatus = 'Not Qualified' THEN 1 ELSE 0 END)
        AS NotQualifiedRoleCount,
    STRING_AGG(CASE WHEN RoleQualificationStatus = 'Qualified' THEN RoleName END, '; ')
        AS QualifiedRoles,
    STRING_AGG(CASE WHEN RoleQualificationStatus = 'Not Qualified' THEN RoleName END, '; ')
        AS NotQualifiedRoles
FROM RoleStatus
GROUP BY EmployeeMasterID, EmployeeName
ORDER BY EmployeeName;

RequiredWAF AS (
    SELECT DISTINCT
        TRY_CAST(TRIM(value) AS BIGINT) AS WorkActivityFunctionID
    FROM vm_cedemo_oqrequirements
    CROSS APPLY STRING_SPLIT(WorkActivityFunctionIDs, ';')
    WHERE DescriptionName = 'Live Mains and Services'
      AND TRIM(value) <> ''
),
EmployeeRequirements AS (
    SELECT
        ce.EmployeeMasterID,
        ce.EmployeeName,
        rw.WorkActivityFunctionID
    FROM ContractorEmployees ce
    CROSS JOIN RequiredWAF rw
),
-- step3: check for qualification
QualificationCheck AS (
    SELECT
        er.EmployeeMasterID,
        er.EmployeeName,
        er.WorkActivityFunctionID,
        CASE
            WHEN q.WorkActivityFunctionID IS NOT NULL
                 AND q.Qualified = 'Yes'
            THEN 1
            ELSE 0
        END AS QualifiedFlag
    FROM EmployeeRequirements er
    LEFT JOIN vm_cedemo_ContractorEmployeesWAFQualifications q
        ON er.EmployeeMasterID = q.EmployeeMasterID
        AND er.WorkActivityFunctionID = q.WorkActivityFunctionID
),
EmployeeStatus AS (
    SELECT
        EmployeeMasterID,
        EmployeeName,
        COUNT(*) AS TotalRequiredWAFs,
        SUM(QualifiedFlag) AS QualifiedWAFs,
        CASE
            WHEN COUNT(*) = SUM(QualifiedFlag)
            THEN 'Qualified'
            ELSE 'Not Qualified'
        END AS EmployeeQualificationStatus
    FROM QualificationCheck
    GROUP BY EmployeeMasterID, EmployeeName
)
-- Employee details
SELECT *
FROM EmployeeStatus
ORDER BY EmployeeName;
-- Summary counts
WITH ContractorEmployees AS (
    SELECT DISTINCT
        EmployeeMasterID,
        EmployeeName
    FROM vm_cedemo_contractoremployees_active
    WHERE IsActive = 1
      AND (ContractorName LIKE '%CAC%' 
           OR ContractorDisplayName LIKE '%CAC%')
),
RequiredWAF AS (
    SELECT DISTINCT
        TRY_CAST(TRIM(value) AS BIGINT) AS WorkActivityFunctionID
    FROM vm_cedemo_oqrequirements
    CROSS APPLY STRING_SPLIT(WorkActivityFunctionIDs, ';')
    WHERE DescriptionName = 'Live Mains and Services'
      AND TRIM(value) <> ''
),
EmployeeRequirements AS (
    SELECT
        ce.EmployeeMasterID,
        ce.EmployeeName,
        rw.WorkActivityFunctionID
    FROM ContractorEmployees ce
    CROSS JOIN RequiredWAF rw
),
QualificationCheck AS (
    SELECT
        er.EmployeeMasterID,
        er.EmployeeName,
        er.WorkActivityFunctionID,
        CASE
            WHEN q.WorkActivityFunctionID IS NOT NULL
                 AND q.Qualified = 'Yes'
            THEN 1
            ELSE 0
        END AS QualifiedFlag
    FROM EmployeeRequirements er
    LEFT JOIN vm_cedemo_ContractorEmployeesWAFQualifications q
        ON er.EmployeeMasterID = q.EmployeeMasterID
        AND er.WorkActivityFunctionID = q.WorkActivityFunctionID
),
EmployeeStatus AS (
    SELECT
        EmployeeMasterID,
        EmployeeName,
        COUNT(*) AS TotalRequiredWAFs,
        SUM(QualifiedFlag) AS QualifiedWAFs,
        CASE
            WHEN COUNT(*) = SUM(QualifiedFlag)
            THEN 'Qualified'
            ELSE 'Not Qualified'
        END AS EmployeeQualificationStatus
    FROM QualificationCheck
    GROUP BY EmployeeMasterID, EmployeeName
)

SELECT
    COUNT(*) AS TotalEmployees,
    SUM(CASE WHEN EmployeeQualificationStatus = 'Qualified' THEN 1 ELSE 0 END) AS TotalQualifiedEmployees,
    SUM(CASE WHEN EmployeeQualificationStatus = 'Not Qualified' THEN 1 ELSE 0 END) AS TotalNotQualifiedEmployees
FROM EmployeeStatus;
